@startuml
actor "Doctor/Admin" as Client
participant UI
participant InventoryController
database Database

Client -> UI: Enter keyword
UI -> InventoryController: Send keyword
InventoryController -> Database: Search Inventory by keyword
Database --> InventoryController: Return list of Inventory
InventoryController --> UI: Return list of Inventory
UI --> Client: Show inventory list

Client -> UI: Select Inventory
UI -> InventoryController: Send Inventory
InventoryController -> Database: Send Inventory ID
Database -> Database: Get Inventory ID

alt Inventory Found
    Database --> InventoryController: Return Inventory Information
    InventoryController --> UI: Send Inventory Information
    UI --> Client: Show Inventory Information
else Inventory Not Found
    Database --> InventoryController: Return Not Found
    InventoryController --> UI: Send Error Message
    UI --> Client: Show Error Message
end
@enduml
